namespace Gaia {

namespace Posix {
using Errno = int;

inline Errno error_to_errno(Error error) {
  switch (error) {
  case Error::NO_SUCH_FILE_OR_DIRECTORY:
    return ENOENT;
  case Error::IS_A_DIRECTORY:
    return EISDIR;
  case Error::NOT_A_DIRECTORY:
    return ENOTDIR;
  case Error::NOT_IMPLEMENTED:
    return ENOSYS;
  case Error::NOT_A_TTY:
    return ENOTTY;
  default:
    return -1;
  }
}
} // namespace Posix
